CREATE TABLE user(
username VARCHAR(30) NOT NULL PRIMARY KEY,
password VARCHAR(50) NOT NULL,
email VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE playlist(
nome VARCHAR(40) PRIMARY KEY,
usernameuser VARCHAR(30) NOT NULL,
genere VARCHAR(30),
pathcoverplaylist VARCHAR(100),
FOREIGN KEY(usernameuser) REFERENCES user(username)
ON DELETE CASCADE
ON UPDATE CASCADE
);

CREATE TABLE contiene(
nomecanzone VARCHAR(30) NOT NULL,
nomeplaylist VARCHAR(30) NOT NULL,
FOREIGN KEY(nomecanzone) REFERENCES canzone(titolo)
ON DELETE CASCADE
ON UPDATE CASCADE,
FOREIGN KEY(nomeplaylist) REFERENCES playlist(nome)
ON DELETE CASCADE
ON UPDATE CASCADE
);

CREATE TABLE canzone(
titolo VARCHAR(30) NOT NULL PRIMARY KEY,
path VARCHAR(100) NOT NULL
);
